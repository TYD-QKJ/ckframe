//ckeditor编辑器插件

var obj;    //定义对象

$(function(){

      //指定高度不要低于560,否则跳出的文件框会显示不完整

      obj = new widget_textarea({height : "560px"});

      //指定替换textarea id=editor1

      obj.init("editor1");    

});
